        public class LeapYear{
            // Main method
            public static void main(String[] args){
                // Create a Scanner
                try (Scanner input = new Scanner(System.in)){
                    int year = input.nextInt();
                    // Check if the year is greater than 1582
                    if(year>=1582){
                        // Check if the year is a leap year
                        if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) System.out.println("Leap Year");

        import java.util.Scanner;
        public class ArmstrongNum {
            // main method
            public static void main(String[] args) {
                // Scanner class to take input from user
                try (Scanner input = new Scanner(System.in)){
                    // Taking input from user
                    int number = input.nextInt();
                    int sum=0;
                    int originalNumber = number;
                    int remainder;
                    // Checking if the number is Armstrong or not
                    while (originalNumber!=0){
                        remainder=originalNumber%10;
                        sum+=remainder*remainder*remainder;
                        originalNumber/=10;
                    }
                    // Printing the result
                    if(number==sum) System.out.println("It is Armstrong");
                    else System.out.println("It is not Armstrong");
                }

        import java.util.Scanner;
        
        
        public class SwitchCase {
            // Function to calculate the result of the operation
            public static void main(String[] args) {
                // Taking input from the user
                try (Scanner input = new Scanner(System.in)){
                    double first = input.nextInt();
                    double second = input.nextInt();
                    String op = input.next();
                    // Calculating the result of the operation
                    if(op.equals("+") || op.equals("-") || op.equals("*") || op.equals("/")){
                        switch (op) {
                            case "+" -> System.out.println(first + second);
                            case "-" -> System.out.println(first - second);
                            case "*" -> System.out.println(first * second);
                            case "/" -> System.out.println(first / second);
                        }
                    }
                    else System.out.println("Invalid Operator");
                }
            }
        }
        
        Write a program DayOfWeek that takes a date as input and prints the day of the week that the date falls on. Your program should take three command-line arguments: m (month), d (day), and y (year). For m use 1 for January, 2 for February, and so forth. For output print 0 for Sunday, 1 for Monday, 2 for Tuesday, and so forth. Use the following formulas, for the Gregorian calendar (where / denotes integer division):
        y0 = y − (14 − m) / 12
        x = y0 + y0/4 − y0/100 + y0/400
        m0 = m + 12 × ((14 − m) / 12) − 2
        d0 = (d + x + 31m0 / 12) mod 7
        
        
        import java.util.Scanner;
        // Enum to store the days of the week
        enum Day {Sunday,Monday,Tuesday,Wednesday,Thursday,Friday,Saturday;}
        
        
        public class DayOfWeek {
            // Function to calculate the day of the week
            public static void main(String[] args) {
                try (Scanner input = new Scanner(System.in)){
                    // Taking input from the user
                    int m = input.nextInt();
                    int d = input.nextInt();
                    int y = input.nextInt();
                    // Calculating the day of the week
                    int y0 = y - (14 - m) / 12;
                    int x = y0 + y0/4 - y0/100 + y0/400;
                    int m0 = m + 12 * ((14 - m) / 12) - 2;
                    int d0 = (d + x + 31*m0 / 12)%7;
                    // Printing the day of the week
                    System.out.print(Day.values()[d0]);
                }
            }
        }
